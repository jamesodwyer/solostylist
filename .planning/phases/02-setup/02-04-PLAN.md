```sql
CREATE TABLE public.client_notes (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_user_id UUID NOT NULL,
  client_id UUID NOT NULL REFERENCES public.clients(id) ON DELETE CASCADE,
  note_type TEXT NOT NULL CHECK (note_type IN ('general', 'colour_formula', 'treatment')),
  content TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE public.colour_formulas (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_user_id UUID NOT NULL,
  client_id UUID NOT NULL REFERENCES public.clients(id) ON DELETE CASCADE,
  formula TEXT NOT NULL,
  notes TEXT,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  updated_at TIMESTAMPTZ NOT NULL DEFAULT now()
);

CREATE TABLE public.tags (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_user_id UUID NOT NULL,
  name TEXT NOT NULL,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE (owner_user_id, name)
);

CREATE TABLE public.client_tags (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_user_id UUID NOT NULL,
  client_id UUID NOT NULL REFERENCES public.clients(id) ON DELETE CASCADE,
  tag_id UUID NOT NULL REFERENCES public.tags(id) ON DELETE CASCADE,
  created_at TIMESTAMPTZ NOT NULL DEFAULT now(),
  UNIQUE (client_id, tag_id)
);
```

**Step 1: Create `src/lib/actions/notes.ts`.**

```typescript
'use server'
import { createClient } from '@/lib/supabase/server'
import { redirect } from 'next/navigation'
import { revalidatePath } from 'next/cache'
import { z } from 'zod'
```

Schemas:
- `noteSchema`: client_id (string uuid), note_type (enum 'general'|'treatment'), content (string min 1)
- `colourFormulaSchema`: client_id (string uuid), formula (string min 1), notes (string optional)

Server Actions:
1. `addNote(data)` — validate, getUser(), insert into client_notes with owner_user_id, revalidatePath(`/clients/${data.client_id}`)
2. `updateNote(id: string, data: { content: string })` — validate content not empty, update where id AND owner_user_id, revalidatePath
3. `deleteNote(id: string, clientId: string)` — delete where id AND owner_user_id, revalidatePath(`/clients/${clientId}`)
4. `addColourFormula(data)` — validate, insert into colour_formulas with owner_user_id, revalidatePath
5. `updateColourFormula(id: string, data: { formula: string; notes?: string })` — validate, update, revalidatePath
6. `deleteColourFormula(id: string, clientId: string)` — delete where id AND owner_user_id, revalidatePath

All actions follow the same pattern: create server client, getUser() (redirect if no user), validate, mutate, revalidate. Return `{ success: true }` or `{ error: string }`.

**Step 2: Create `src/lib/actions/tags.ts`.**

Schemas:
- `tagSchema`: name (string min 1)
- `clientTagSchema`: client_id (string uuid), tag_id (string uuid)

Server Actions:
1. `createTag(name: string)` — getUser(), upsert into tags (owner_user_id + name, onConflict: 'owner_user_id,name'), return the tag `{ data: tag }` so the picker can use it immediately. Use `.select().single()` after upsert to get the row.
2. `addTagToClient(clientId: string, tagId: string)` — getUser(), insert into client_tags (owner_user_id, client_id, tag_id), handle unique constraint violation gracefully (already tagged = success), revalidatePath(`/clients/${clientId}`)
3. `removeTagFromClient(clientId: string, tagId: string)` — getUser(), delete from client_tags where client_id AND tag_id AND owner_user_id, revalidatePath(`/clients/${clientId}`)

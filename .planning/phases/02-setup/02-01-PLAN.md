```typescript
/** Format integer pennies to £ string. E.g. 1500 → "£15.00" */
export function formatPrice(pennies: number): string {
  return `£${(pennies / 100).toFixed(2)}`
}

/** Parse a decimal string to integer pennies. E.g. "15.00" → 1500 */
export function parsePriceToPennies(input: string): number {
  const value = parseFloat(input)
  if (isNaN(value)) return 0
  return Math.round(value * 100)
}
```

**Key constraints:**

**Step 1: Create Server Action `src/lib/actions/profile.ts`.**

```typescript
'use server'

import { createClient } from '@/lib/supabase/server'
import { redirect } from 'next/navigation'
import { revalidatePath } from 'next/cache'
import { z } from 'zod'
import type { WorkingHours } from '@/lib/types/database'
```

Define schemas:
- `dayScheduleSchema`: object with `enabled` (boolean), `start` (string), `end` (string)
- `workingHoursSchema`: object with keys mon-sun, each dayScheduleSchema
- `onboardingSchema`: object with `trading_name` (string, min 1, "Trading name is required"), `phone` (string, optional, default ''), `working_hours` (workingHoursSchema), `default_slot_minutes` (number, int, one of 15/30/45/60 via z.union or z.enum mapped)

Export `async function completeOnboarding(data: z.infer<typeof onboardingSchema>)`:
1. Create Supabase server client
2. Get user via `getUser()` — if no user, `redirect('/login')`
3. `safeParse(data)` — if invalid, return `{ error: parsed.error.flatten().fieldErrors }`
4. Update profiles table: `supabase.from('profiles').update({ trading_name, phone, working_hours, default_slot_minutes, onboarding_completed: true, updated_at: new Date().toISOString() }).eq('id', user.id)`
5. If update error, return `{ error: { _form: [error.message] } }`
6. Seed preset tags for this user (idempotent upsert):
   ```typescript
   const PRESET_TAGS = ['Allergy', 'VIP', 'New client', 'Sensitive scalp', 'Regular']
   await supabase.from('tags').upsert(
     PRESET_TAGS.map(name => ({ owner_user_id: user.id, name })),
     { onConflict: 'owner_user_id,name', ignoreDuplicates: true }
   )
   ```
7. `revalidatePath('/')` then `redirect('/diary')`

Also export `async function updateProfile(data: Partial<z.infer<typeof onboardingSchema>>)` for the Settings page to reuse later. Same pattern but without `onboarding_completed` or tag seeding, and `revalidatePath('/settings')` instead of redirect.
